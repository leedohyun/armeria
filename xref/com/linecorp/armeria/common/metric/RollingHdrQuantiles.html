55      private static final double[] PREDEFINED_QUANTILES = { 0, 0.5, 0.75, 0.9, 0.95, 0.98, 0.99, 0.999, 1.0 };
56      private static final Collection<Double> MONITORED = Arrays.stream(PREDEFINED_QUANTILES).boxed()
57                                                                .collect(toImmutableList());
58  
59      @VisibleForTesting
60      static final int NUM_SIGNIFICANT_VALUE_DIGITS = 2;
61  
62      /**
63       * A value less than this value will be replaced so that it does not exceed the dynamic range limits of
64       * {@link DoubleRecorder}.
65       */
66      @VisibleForTesting
67      static final double MIN_VALUE = 1.0 / TimeUnit.SECONDS.toMillis(1);
68  
69      /**
70       * A value greater than this value will be replaced so that it does not exceed the dynamic range limits of
71       * {@link DoubleRecorder}.
72       */
73      @VisibleForTesting
74      static final double MAX_VALUE = 70368744177663L;
75  
76      private final long intervalBetweenResettingMillis;
77      private final long creationTimestamp;
78      private final ArchivedHistogram[] archive;
79      private final boolean historySupported;
80      private final LongSupplier clock;
81      private final DoubleHistogram temporarySnapshotHistogram;
82  
83      private final Phase left;
84      private final Phase right;
85      private final Phase[] phases;
86      private final AtomicReference<Phase> currentPhaseRef;
87  
88      private double[] snapshot;
89      private long lastSnapshotTimeMillis;
90  
91      RollingHdrQuantiles() {
92          this(5, Duration.ofMinutes(2).toMillis(), System::currentTimeMillis);
93      }
94  
95      RollingHdrQuantiles(int numberHistoryChunks, long intervalBetweenResettingMillis) {
96          this(numberHistoryChunks, intervalBetweenResettingMillis, System::currentTimeMillis);
97      }
98  
99      RollingHdrQuantiles(int numberHistoryChunks, long intervalBetweenResettingMillis,
100                         LongSupplier clock) {
101 
102         this.intervalBetweenResettingMillis = intervalBetweenResettingMillis;
103         this.clock = clock;
104         creationTimestamp = clock.getAsLong();
105 
106         left = new Phase(creationTimestamp + intervalBetweenResettingMillis);
107         right = new Phase(Long.MAX_VALUE);
108         phases = new Phase[] { left, right };
109         currentPhaseRef = new AtomicReference<>(left);
110 
111         historySupported = numberHistoryChunks > 0;
112         if (historySupported) {
113             archive = new ArchivedHistogram[numberHistoryChunks];
114             for (int i = 0; i < numberHistoryChunks; i++) {
115                 final DoubleHistogram archivedHistogram = createNonConcurrentCopy(left.intervalHistogram);
116                 archive[i] = new ArchivedHistogram(archivedHistogram, Long.MIN_VALUE);
117             }
118         } else {
119             archive = null;
120         }
121 
122         temporarySnapshotHistogram = createNonConcurrentCopy(left.intervalHistogram);
123     }
124 
125     @Override
126     public void observe(double value) {
127         observe(value, 0);
128     }
129 
130     public void observe(double value, double expectedIntervalBetweenValueSamples) {
131         value = Math.max(Math.min(value, MAX_VALUE), MIN_VALUE);
132 
133         final long currentTimeMillis = clock.getAsLong();
134         final Phase currentPhase = currentPhaseRef.get();
135         if (currentTimeMillis < currentPhase.proposedInvalidationTimestamp) {
136             currentPhase.recorder.recordValueWithExpectedInterval(value, expectedIntervalBetweenValueSamples);
137             return;
138         }
139 
140         final Phase nextPhase = currentPhase == left ? right : left;
141         nextPhase.recorder.recordValueWithExpectedInterval(value, expectedIntervalBetweenValueSamples);
142 
143         if (!currentPhaseRef.compareAndSet(currentPhase, nextPhase)) {
144             // another writer achieved progress and must submit rotation task to backgroundExecutor
145             return;
146         }
147 
148         // Current thread is responsible to rotate phases.
149         rotate(currentTimeMillis, currentPhase, nextPhase);
150     }
151 
152     private synchronized void rotate(long currentTimeMillis, Phase currentPhase, Phase nextPhase) {
153         try {
154             currentPhase.intervalHistogram =
155                     currentPhase.recorder.getIntervalHistogram(currentPhase.intervalHistogram);
156             addSecondToFirst(currentPhase.totalsHistogram, currentPhase.intervalHistogram);
157             if (historySupported) {
158                 // move values from recorder to correspondent archived histogram
159                 final long currentPhaseNumber =
160                         (currentPhase.proposedInvalidationTimestamp - creationTimestamp) /
161                         intervalBetweenResettingMillis;
162                 final int correspondentArchiveIndex = (int) ((currentPhaseNumber - 1) % archive.length);
163                 final ArchivedHistogram correspondentArchivedHistogram = archive[correspondentArchiveIndex];
164                 reset(correspondentArchivedHistogram.histogram);
165                 addSecondToFirst(correspondentArchivedHistogram.histogram, currentPhase.totalsHistogram);
166                 correspondentArchivedHistogram.proposedInvalidationTimestamp =
167                         currentPhase.proposedInvalidationTimestamp +
168                         archive.length * intervalBetweenResettingMillis;
169             }
170             reset(currentPhase.totalsHistogram);
171         } finally {
172             final long millisSinceCreation = currentTimeMillis - creationTimestamp;
173             final long intervalsSinceCreation = millisSinceCreation / intervalBetweenResettingMillis;
174             currentPhase.proposedInvalidationTimestamp = Long.MAX_VALUE;
175             nextPhase.proposedInvalidationTimestamp =
176                     creationTimestamp + (intervalsSinceCreation + 1) * intervalBetweenResettingMillis;
177         }
178     }

180     @Override
181     public Double get(double percentile) {
182         final double[] snapshot = getSnapshot();
183         for (int i = 0; i < PREDEFINED_QUANTILES.length; i++) {
184             if (percentile <= PREDEFINED_QUANTILES[i]) {
185                 return snapshot[i];
186             }
187         }
188         return snapshot[PREDEFINED_QUANTILES.length - 1];
189     }
190 
191     @Override
192     public Collection<Double> monitored() {
193         return MONITORED;
194     }
195 
196     private synchronized double[] getSnapshot() {
197         final long currentTimeMillis = clock.getAsLong();
198         final double[] oldSnapshot = snapshot;
199         if (oldSnapshot != null && currentTimeMillis - lastSnapshotTimeMillis < 1000) {
200             return oldSnapshot;
201         }
202 
203         reset(temporarySnapshotHistogram);
204 
205         for (Phase phase : phases) {
206             if (phase.isNeedToBeReportedToSnapshot(currentTimeMillis)) {
207                 phase.intervalHistogram = phase.recorder.getIntervalHistogram(phase.intervalHistogram);
208                 addSecondToFirst(phase.totalsHistogram, phase.intervalHistogram);
209                 addSecondToFirst(temporarySnapshotHistogram, phase.totalsHistogram);
210             }
211         }
212         if (historySupported) {
213             for (ArchivedHistogram archivedHistogram : archive) {
214                 if (archivedHistogram.proposedInvalidationTimestamp > currentTimeMillis) {
215                     addSecondToFirst(temporarySnapshotHistogram, archivedHistogram.histogram);
216                 }
217             }
218         }
219 
220         final double[] snapshot = new double[PREDEFINED_QUANTILES.length];
221         for (int i = 0; i < PREDEFINED_QUANTILES.length; i++) {
222             final double percentile = PREDEFINED_QUANTILES[i] * 100.0;
223             snapshot[i] = temporarySnapshotHistogram.getValueAtPercentile(percentile);
224         }
225         this.snapshot = snapshot;
226         lastSnapshotTimeMillis = currentTimeMillis;
227         return snapshot;
228     }
229 
230     @VisibleForTesting
231     synchronized void discardCachedSnapshot() {
232         snapshot = null;
233     }
234 
235     @VisibleForTesting
236     int estimatedFootprintInBytes() {
237         // each histogram has equivalent pessimistic estimation
238         int oneHistogramPessimisticFootprint = temporarySnapshotHistogram.getEstimatedFootprintInBytes();
239 
240         // 4 - two recorders with two histogram
241         // 2 - two histogram for storing accumulated values from current phase
242         // 1 - temporary histogram used for snapshot extracting
243         return oneHistogramPessimisticFootprint * ((archive != null ? archive.length : 0) + 4 + 2 + 1);
244     }
245 
246     private static void reset(DoubleHistogram histogram) {
247         if (histogram.getTotalCount() > 0) {
248             histogram.reset();
249         }
250     }
251 
252     private static void addSecondToFirst(DoubleHistogram first, DoubleHistogram second) {
253         if (second.getTotalCount() > 0) {
254             first.add(second);
255         }
256     }
257 
258     private static DoubleHistogram createNonConcurrentCopy(DoubleHistogram source) {
259         final DoubleHistogram copy = new DoubleHistogram(
260                 source.getHighestToLowestValueRatio(),
261                 source.getNumberOfSignificantValueDigits());
262         copy.setAutoResize(source.isAutoResize());
263         return copy;
264     }
265 
266     private static String histogramToString(DoubleHistogram histogram) {
267         try (ByteArrayOutputStream baos = new ByteArrayOutputStream()) {
268             final PrintStream writer = new PrintStream(baos);
269             histogram.outputPercentileDistribution(writer, 1.0);
270             return new String(baos.toByteArray());
271         } catch (IOException e) {
272             throw new IllegalStateException(e);
273         }
274     }
275 
276     private static String printArray(Object[] array, String elementName) {
277         final StringBuilder buf = new StringBuilder();
278         buf.append('{');
279         for (int i = 0; i < array.length; i++) {
280             buf.append('\n')
281                .append(elementName)
282                .append('[').append(i).append("]=")
283                .append(array[i]);
284         }
285         buf.append("\n}");
286         return buf.toString();
287     }

289     private static final class ArchivedHistogram {
290 
291         final DoubleHistogram histogram;
292         volatile long proposedInvalidationTimestamp;
293 
294         ArchivedHistogram(DoubleHistogram histogram, long proposedInvalidationTimestamp) {
295             this.histogram = histogram;
296             this.proposedInvalidationTimestamp = proposedInvalidationTimestamp;
297         }
298 
299         @Override
300         public String toString() {
301             return "ArchivedHistogram{" +
302                     "\n, proposedInvalidationTimestamp=" + proposedInvalidationTimestamp +
303                     "\n, histogram=" + histogramToString(histogram) +
304                     "\n}";
305         }
306     }
307 
308     private final class Phase {
309 
310         final DoubleRecorder recorder;
311         final DoubleHistogram totalsHistogram;
312         DoubleHistogram intervalHistogram;
313         volatile long proposedInvalidationTimestamp;
314 
315         Phase(long proposedInvalidationTimestamp) {
316             recorder = new DoubleRecorder(NUM_SIGNIFICANT_VALUE_DIGITS);
317             intervalHistogram = recorder.getIntervalHistogram();
318             intervalHistogram.setAutoResize(true);
319             totalsHistogram = intervalHistogram.copy();
320             totalsHistogram.setAutoResize(true);
321             this.proposedInvalidationTimestamp = proposedInvalidationTimestamp;
322         }
323 
324         @Override
325         public String toString() {
326             return "Phase{" +
327                     "\n, proposedInvalidationTimestamp=" + proposedInvalidationTimestamp +
328                     "\n, totalsHistogram=" + (totalsHistogram != null ? histogramToString(totalsHistogram)
329                                                                       : "null") +
330                     "\n, intervalHistogram=" + histogramToString(intervalHistogram) +
331                     "\n}";
332         }
333 
334         boolean isNeedToBeReportedToSnapshot(long currentTimeMillis) {
335             final long proposedInvalidationTimestampLocal = proposedInvalidationTimestamp;
336             if (proposedInvalidationTimestampLocal > currentTimeMillis) {
337                 return true;
338             }
339             if (!historySupported) {
340                 return false;
341             }
342             final long correspondentChunkProposedInvalidationTimestamp =
343                     proposedInvalidationTimestampLocal + archive.length * intervalBetweenResettingMillis;
344             return correspondentChunkProposedInvalidationTimestamp > currentTimeMillis;
345         }
346     }
347 
348     @Override
349     public String toString() {
350         return "RollingHdrQuantiles{" +
351                 "\nintervalBetweenResettingMillis=" + intervalBetweenResettingMillis +
352                 ",\n creationTimestamp=" + creationTimestamp +
353                 (!historySupported ? "" : ",\n archive=" + printArray(archive, "chunk")) +
354                 ",\n left=" + left +
355                 ",\n right=" + right +
356                 ",\n currentPhase=" + (currentPhaseRef.get() == left ? "left" : "right") +
357                 ",\n temporarySnapshotHistogram=" + histogramToString(temporarySnapshotHistogram)  +
358                 '}';
359     }
